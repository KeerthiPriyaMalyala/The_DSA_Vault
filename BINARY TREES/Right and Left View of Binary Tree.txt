Left View of Binary Tree 🌳
❓ What is Left View?

Nodes visible when tree is viewed from the left side.

Take the first node of every level.

🌟 Example
           1
         /   \
        2     3
       / \     \
      4   5     6
Left View
1 2 4
⚡ Easy BFS Approach
💡 Idea
Use Queue for Level Order Traversal.
At every level:
first node = left view.
💻 Java Code
import java.util.*;

class Node {
    int data;
    Node left, right;

    Node(int data) {
        this.data = data;
    }
}

public class Main {

    static void leftView(Node root) {

        if (root == null)
            return;

        Queue<Node> q = new LinkedList<>();

        q.add(root);

        while (!q.isEmpty()) {

            int size = q.size();

            for (int i = 0; i < size; i++) {

                Node curr = q.poll();

                // first node of level
                if (i == 0)
                    System.out.print(curr.data + " ");

                if (curr.left != null)
                    q.add(curr.left);

                if (curr.right != null)
                    q.add(curr.right);
            }
        }
    }

    public static void main(String[] args) {

        Node root = new Node(1);

        root.left = new Node(2);
        root.right = new Node(3);

        root.left.left = new Node(4);
        root.left.right = new Node(5);

        root.right.right = new Node(6);

        leftView(root);
    }
}
▶️ Output
1 2 4
🌳 Right View of Binary Tree
❓ What is Right View?

Nodes visible from the right side.

Take the last node of every level.

💻 Java Code
import java.util.*;

class Node {
    int data;
    Node left, right;

    Node(int data) {
        this.data = data;
    }
}

public class Main {

    static void rightView(Node root) {

        if (root == null)
            return;

        Queue<Node> q = new LinkedList<>();

        q.add(root);

        while (!q.isEmpty()) {

            int size = q.size();

            for (int i = 0; i < size; i++) {

                Node curr = q.poll();

                // last node of level
                if (i == size - 1)
                    System.out.print(curr.data + " ");

                if (curr.left != null)
                    q.add(curr.left);

                if (curr.right != null)
                    q.add(curr.right);
            }
        }
    }

    public static void main(String[] args) {

        Node root = new Node(1);

        root.left = new Node(2);
        root.right = new Node(3);

        root.left.left = new Node(4);
        root.left.right = new Node(5);

        root.right.right = new Node(6);

        rightView(root);
    }
}
▶️ Output
1 3 6
🧠 Easy Trick
Left View  → first node of level
Right View → last node of level
i == 0         → Left View
i == size - 1  → Right View 😄
